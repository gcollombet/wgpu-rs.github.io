11/// Open connection to a graphics and/or compute device.
12///
13/// Responsible for the creation of most rendering and compute resources.
14/// These are then used in commands, which are submitted to a [`Queue`].
15///
16/// A device may be requested from an adapter with [`Adapter::request_device`].
17///
18/// Corresponds to [WebGPU `GPUDevice`](https://gpuweb.github.io/gpuweb/#gpu-device).
19#[derive(Debug, Clone)]
20pub struct Device {
21    pub(crate) inner: dispatch::DispatchDevice,
22}
23#[cfg(send_sync)]
24static_assertions::assert_impl_all!(Device: Send, Sync);

26crate::cmp::impl_eq_ord_hash_proxy!(Device => .inner);

52    /// Constructs a stub device for testing using [`Backend::Noop`].
53    ///
54    /// This is a convenience function which avoids the configuration, `async`, and fallibility
55    /// aspects of constructing a device through `Instance`.
56    #[cfg(feature = "noop")]
57    pub fn noop(desc: &DeviceDescriptor<'_>) -> (Device, Queue) {
58        use core::future::Future as _;
59        use core::pin::pin;
60        use core::task;
61        let ctx = &mut task::Context::from_waker(waker::noop_waker_ref());
62
63        let instance = Instance::new(&InstanceDescriptor {
64            backends: Backends::NOOP,
65            backend_options: BackendOptions {
66                noop: NoopBackendOptions { enable: true },
67                ..Default::default()
68            },
69            ..Default::default()
70        });
71
72        // Both of these futures are trivial and should complete instantaneously,
73        // so we do not need an executor and can just poll them once.
74        let task::Poll::Ready(Ok(adapter)) =
75            pin!(instance.request_adapter(&RequestAdapterOptions::default())).poll(ctx)
76        else {
77            unreachable!()
78        };
79        let task::Poll::Ready(Ok(device_and_queue)) = pin!(adapter.request_device(desc)).poll(ctx)
80        else {
81            unreachable!()
82        };
83        device_and_queue
84    }

86    /// Check for resource cleanups and mapping callbacks. Will block if [`PollType::Wait`] is passed.
87    ///
88    /// Return `true` if the queue is empty, or `false` if there are more queue
89    /// submissions still in flight. (Note that, unless access to the [`Queue`] is
90    /// coordinated somehow, this information could be out of date by the time
91    /// the caller receives it. `Queue`s can be shared between threads, so
92    /// other threads could submit new work at any time.)
93    ///
94    /// When running on WebGPU, this is a no-op. `Device`s are automatically polled.
95    pub fn poll(&self, poll_type: PollType) -> Result<crate::PollStatus, crate::PollError> {
96        self.inner.poll(poll_type.map_index(|s| s.index))
97    }

554    ///
555    /// This can be passed to [`Device::create_compute_pipeline`]
556    /// and [`Device::create_render_pipeline`] to either accelerate these
557    /// or add the cache results from those.
558    ///
559    /// # Safety
560    ///
561    /// If the `data` field of `desc` is set, it must have previously been returned from a call
562    /// to [`PipelineCache::get_data`][^saving]. This `data` will only be used if it came
563    /// from an adapter with the same [`util::pipeline_cache_key`].
564    /// This *is* compatible across wgpu versions, as any data format change will
565    /// be accounted for.
566    ///
567    /// It is *not* supported to bring caches from previous direct uses of backend APIs
568    /// into this method.
569    ///
570    /// # Errors
571    ///
572    /// Returns an error value if:
573    ///  * the [`PIPELINE_CACHE`](wgt::Features::PIPELINE_CACHE) feature is not enabled
574    ///  * this device is invalid; or
575    ///  * the device is out of memory
576    ///
577    /// This method also returns an error value if:
578    ///  * The `fallback` field on `desc` is false; and
579    ///  * the `data` provided would not be used[^data_not_used]
580    ///
581    /// If an error value is used in subsequent calls, default caching will be used.
582    ///
583    /// [^saving]: We do recognise that saving this data to disk means this condition
584    /// is impossible to fully prove. Consider the risks for your own application in this case.
585    ///
586    /// [^data_not_used]: This data may be not used if: the data was produced by a prior
587    /// version of wgpu; or was created for an incompatible adapter, or there was a GPU driver
588    /// update. In some cases, the data might not be used and a real value is returned,
589    /// this is left to the discretion of GPU drivers.
590    #[must_use]

735/// Lower level source of the error.
736///
737/// `Send + Sync` varies depending on configuration.
738#[cfg(not(send_sync))]
739#[cfg_attr(docsrs, doc(cfg(all())))]

742/// Errors resulting from usage of GPU APIs.
743///
744/// By default, errors translate into panics. Depending on the backend and circumstances,
745/// errors may occur synchronously or asynchronously. When errors need to be handled, use
746/// [`Device::push_error_scope()`] or [`Device::on_uncaptured_error()`].
747#[derive(Debug)]
748pub enum Error {
749    /// Out of memory.
750    OutOfMemory {
751        /// Lower level source of the error.
752        source: ErrorSource,
753    },
754    /// Validation error, signifying a bug in code or data provided to `wgpu`.
755    Validation {
756        /// Lower level source of the error.
757        source: ErrorSource,
758        /// Description of the validation error.
759        description: String,
760    },
761    /// Internal error. Used for signalling any failures not explicitly expected by WebGPU.
762    ///
763    /// These could be due to internal implementation or system limits being reached.
764    Internal {
765        /// Lower level source of the error.
766        source: ErrorSource,
767        /// Description of the internal GPU error.
768        description: String,
769    },
770}
771#[cfg(send_sync)]
772static_assertions::assert_impl_all!(Error: Send, Sync);
773
774impl error::Error for Error {
775    fn source(&self) -> Option<&(dyn error::Error + 'static)> {
776        match self {
777            Error::OutOfMemory { source } => Some(source.as_ref()),
778            Error::Validation { source, .. } => Some(source.as_ref()),
779            Error::Internal { source, .. } => Some(source.as_ref()),
780        }
781    }
782}
783
784impl fmt::Display for Error {
785    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
786        match self {
787            Error::OutOfMemory { .. } => f.write_str("Out of Memory"),
788            Error::Validation { description, .. } => f.write_str(description),
789            Error::Internal { description, .. } => f.write_str(description),
790        }
791    }
792}

796#[cfg(feature = "noop")]
797mod waker {
798    use core::ptr::null;
799    use core::task::{RawWaker, RawWakerVTable, Waker};
800
801    unsafe fn noop_clone(_data: *const ()) -> RawWaker {
802        noop_raw_waker()
803    }
804
805    unsafe fn noop(_data: *const ()) {}
806
807    const NOOP_WAKER_VTABLE: RawWakerVTable = RawWakerVTable::new(noop_clone, noop, noop, noop);
808
809    const fn noop_raw_waker() -> RawWaker {
810        RawWaker::new(null(), &NOOP_WAKER_VTABLE)
811    }
812
813    /// Get a static reference to a [`Waker`] which
814    /// does nothing when `wake()` is called on it.
815    #[inline]
816    pub fn noop_waker_ref() -> &'static Waker {
817        struct SyncRawWaker(RawWaker);
818        unsafe impl Sync for SyncRawWaker {}
819
820        static NOOP_WAKER_INSTANCE: SyncRawWaker = SyncRawWaker(noop_raw_waker());
821
822        // SAFETY: `Waker` is #[repr(transparent)] over its `RawWaker`.
823        unsafe { &*(&NOOP_WAKER_INSTANCE.0 as *const RawWaker as *const Waker) }
824    }
